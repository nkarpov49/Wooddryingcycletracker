/**
 * Автоматический триггер при редактировании таблицы
 * Запускается при добавлении/изменении строки
 */
function onEdit(e) {
  const sheet = e.source.getActiveSheet();
  
  // Работаем только с листом "events"
  if (sheet.getName() !== SHEET_NAME) {
    return;
  }
  
  const row = e.range.getRow();
  
  // Пропускаем заголовок
  if (row === 1) {
    return;
  }
  
  // Если строка заполнена - синхронизируем
  if (isRowComplete(sheet, row)) {
    Logger.log(`✏️ Строка ${row} заполнена - запуск синхронизации`);
    processRow(sheet, row);
  }
}

/**
 * Синхронизация последней строки из меню
 */
function syncLastRowFromMenu() {
  const ui = SpreadsheetApp.getUi();
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  
  if (!sheet) {
    ui.alert('❌ Ошибка', `Лист "${SHEET_NAME}" не найден!`, ui.ButtonSet.OK);
    return;
  }
  
  const lastRow = sheet.getLastRow();
  
  if (lastRow <= 1) {
    ui.alert('⚠️ Внимание', 'Нет данных для синхронизации', ui.ButtonSet.OK);
    return;
  }
  
  // Проверяем что строка заполнена
  if (!isRowComplete(sheet, lastRow)) {
    ui.alert('⚠️ Внимание', 'Последняя строка заполнена не полностью!', ui.ButtonSet.OK);
    return;
  }
  
  // Отправляем
  Logger.log(`📤 Синхронизация строки ${lastRow} из меню`);
  processRow(sheet, lastRow);
  
  ui.alert('✅ Успешно', `Строка ${lastRow} синхронизирована!\nПроверьте приложение.`, ui.ButtonSet.OK);
}

/**
 * Проверяет что все обязательные поля заполнены
 */
function isRowComplete(sheet, row) {
  const chamberNumber = sheet.getRange(row, COLUMNS.CHAMBER_NUMBER).getValue();
  const woodType = sheet.getRange(row, COLUMNS.WOOD_TYPE).getValue();
  const oldSeqNum = sheet.getRange(row, COLUMNS.OLD_SEQUENTIAL_NUMBER).getValue();
  const oldEndDate = sheet.getRange(row, COLUMNS.OLD_END_DATE).getValue();
  const newSeqNum = sheet.getRange(row, COLUMNS.NEW_SEQUENTIAL_NUMBER).getValue();
  const newStartDate = sheet.getRange(row, COLUMNS.NEW_START_DATE).getValue();
  
  return chamberNumber !== '' && woodType !== '' && oldSeqNum !== '' && 
         oldEndDate !== '' && newSeqNum !== '' && newStartDate !== '';
}

/**
 * Обрабатывает строку и отправляет на сервер
 */
function processRow(sheet, row) {
  try {
    const data = {
      rowNumber: row,
      chamberNumber: parseInt(sheet.getRange(row, COLUMNS.CHAMBER_NUMBER).getValue()),
      woodTypeLithuanian: sheet.getRange(row, COLUMNS.WOOD_TYPE).getValue().toString().trim(),
      oldSequentialNumber: sheet.getRange(row, COLUMNS.OLD_SEQUENTIAL_NUMBER).getValue().toString().trim(),
      oldCycleEndDate: formatDate(sheet.getRange(row, COLUMNS.OLD_END_DATE).getValue()),
      newSequentialNumber: sheet.getRange(row, COLUMNS.NEW_SEQUENTIAL_NUMBER).getValue().toString().trim(),
      newCycleStartDate: formatDate(sheet.getRange(row, COLUMNS.NEW_START_DATE).getValue())
    };
    
    // Валидация
    if (data.chamberNumber < 1 || data.chamberNumber > 21) {
      Logger.log(`❌ Некорректный номер камеры: ${data.chamberNumber}`);
      return;
    }
    
    if (!/^\d{4}$/.test(data.oldSequentialNumber) || !/^\d{4}$/.test(data.newSequentialNumber)) {
      Logger.log(`❌ Порядковые номера должны быть 4-значными`);
      return;
    }
    
    Logger.log(`📤 Отправка данных: камера ${data.chamberNumber}, цикл ${data.oldSequentialNumber} → ${data.newSequentialNumber}`);
    
    // Отправка на сервер
    const response = UrlFetchApp.fetch(URL_WEBHOOK, {
      method: 'post',
      contentType: 'application/json',
      headers: { 'Authorization': `Bearer ${AUTH_TOKEN}` },
      payload: JSON.stringify(data),
      muteHttpExceptions: true
    });
    
    const result = JSON.parse(response.getContentText());
    
    if (result.success) {
      Logger.log(`✅ Строка ${row} обработана успешно`);
    } else if (result.skipped) {
      Logger.log(`⚠️ Строка ${row} уже была обработана ранее`);
    } else {
      Logger.log(`⚠️ Неожиданный ответ: ${response.getContentText()}`);
    }
    
  } catch (error) {
    Logger.log(`❌ Ошибка обработки строки ${row}: ${error.message}`);
  }
}

/**
 * Конвертирует дату в ISO 8601 формат
 */
function formatDate(date) {
  if (date instanceof Date) {
    return Utilities.formatDate(date, "GMT", "yyyy-MM-dd'T'HH:mm:ss'Z'");
  }
  
  if (typeof date === 'string' && date !== '') {
    if (/^\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}$/.test(date.trim())) {
      return date.trim().replace(' ', 'T') + 'Z';
    }
    
    const parsed = new Date(date);
    if (!isNaN(parsed.getTime())) {
      return Utilities.formatDate(parsed, "GMT", "yyyy-MM-dd'T'HH:mm:ss'Z'");
    }
  }
  
  return new Date().toISOString();
}

/**
 * 1. Настройка автоматического триггера
 * Запустите ОДИН РАЗ после установки скрипта
 */
function SetupAutoSync() {
  // Удаляем все старые триггеры
  const triggers = ScriptApp.getProjectTriggers();
  triggers.forEach(trigger => ScriptApp.deleteTrigger(trigger));
  
  Logger.log('=== НАСТРОЙКА АВТОМАТИЧЕСКОЙ СИНХРОНИЗАЦИИ ===');
  Logger.log('Старые триггеры удалены');
  
  // Создаём триггер по ВРЕМЕНИ (каждую минуту)
  ScriptApp.newTrigger('autoSyncOnTime')
    .timeBased()
    .everyMinutes(1)
    .create();
  
  Logger.log('✅ Создан триггер: autoSyncOnTime (каждую минуту)');
  Logger.log('');
  Logger.log('🎉 Автоматическая синхронизация настроена!');
  Logger.log('Новые строки будут синхронизироваться автоматически каждую минуту');
}

/**
 * Автоматическая синхронизация по времени
 * Проверяет новые строки каждую минуту
 */
function autoSyncOnTime() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  
  if (!sheet) {
    Logger.log(`❌ Лист "${SHEET_NAME}" не найден!`);
    return;
  }
  
  const lastRow = sheet.getLastRow();
  
  // Нет данных
  if (lastRow <= 1) {
    return;
  }
  
  // Проверяем только последнюю строку (самая свежая)
  if (isRowComplete(sheet, lastRow)) {
    // Проверяем, была ли уже обработана
    const processed = PropertiesService.getScriptProperties().getProperty(`processed_${lastRow}`);
    
    if (!processed) {
      Logger.log(`⏰ Триггер по времени: обнаружена новая строка ${lastRow}`);
      processRow(sheet, lastRow);
      
      // Помечаем как обработанную
      PropertiesService.getScriptProperties().setProperty(`processed_${lastRow}`, new Date().toISOString());
    }
  }
}

/**
 * 3. Синхронизация всех строк со статусом NEW
 * Обрабатывает все необработанные строки
 */
function SyncAllNew() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  
  if (!sheet) {
    Logger.log(`❌ Лист "${SHEET_NAME}" не найден!`);
    return;
  }
  
  const data = sheet.getDataRange().getValues();
  let processedCount = 0;
  let skippedCount = 0;
  
  Logger.log('=== МАССОВАЯ СИНХРОНИЗАЦИЯ ===');
  Logger.log(`Всего строк: ${data.length - 1}`);
  
  // Пропускаем заголовок
  for (let i = 1; i < data.length; i++) {
    const row = data[i];
    const rowNumber = i + 1;
    
    // Проверяем что строка заполнена
    if (!row[1] || !row[2] || !row[4] || !row[6]) {
      skippedCount++;
      continue;
    }
    
    Logger.log(`\n--- Обработка строки ${rowNumber} ---`);
    processRow(sheet, rowNumber);
    processedCount++;
    
    // Задержка между запросами
    Utilities.sleep(500);
  }
  
  Logger.log('\n=== ИТОГО ===');
  Logger.log(`✅ Обработано: ${processedCount}`);
  Logger.log(`⏭️ Пропущено: ${skippedCount}`);
}

/**
 * 5. Посмотреть логи синхронизации
 * Показывает последние 5 операций из Supabase
 */
function ViewLogs() {
  Logger.log('📋 Получение логов синхронизации из Supabase...');
  
  const logsUrl = URL_WEBHOOK.replace('/sheets/process-row', '/sheets/sync-logs?limit=5');
  
  try {
    const response = UrlFetchApp.fetch(logsUrl, {
      method: 'get',
      headers: { 'Authorization': `Bearer ${AUTH_TOKEN}` },
      muteHttpExceptions: true
    });
    
    const result = JSON.parse(response.getContentText());
    
    Logger.log(`\n=== ЛОГИ СИНХРОНИЗАЦИИ (${result.count} записей) ===\n`);
    
    if (result.count === 0) {
      Logger.log('📭 Логов пока нет');
      return;
    }
    
    result.logs.forEach((log, i) => {
      Logger.log(`\n--- ЛОГ ${i + 1} (${log.timestamp}) ---`);
      Logger.log(`Всего операций: ${log.logs.length}`);
      log.logs.forEach(op => {
        Logger.log(`\n  ${op.action}:`);
        Logger.log(`    Успех: ${op.success ? '✅' : '❌'}`);
        Logger.log(`    Сообщение: ${op.message}`);
        if (op.details) {
          Logger.log(`    Детали: ${JSON.stringify(op.details)}`);
        }
      });
    });
    
  } catch (error) {
    Logger.log(`❌ Ошибка: ${error.message}`);
  }
}

  Logger.log('');

    Logger.log('⚠️ Есть лишние триггеры! Запустите CleanupTriggers()');
  }
